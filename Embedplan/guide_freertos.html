2
FreeRTOS 核心概念（先看懂这些）
15 分钟阅读
什么是 RTOS？

Real-Time Operating System = 实时操作系统。FreeRTOS 让你在单片机上同时"运行"多个任务，本质是快速切换（任务调度器每 1ms 切换一次，速度快到看起来像并行）。

四个核心概念：

  • Task（任务）：就是一个无限循环的函数，有自己的栈空间和优先级。类比：进程
  • Scheduler（调度器）：决定哪个任务现在运行。FreeRTOS 默认抢占式调度，高优先级任务随时可以打断低优先级
  • Queue（队列）：任务间传数据的管道，线程安全，先进先出
  • Semaphore / Mutex：同步机制，防止两个任务同时访问同一个资源（如 SPI 总线）
Task A (优先级 2) ──────┐ ├── Scheduler ──→ CPU 执行 Task B (优先级 1) ──────┘ Task A 等待 → B 运行 → A 就绪 → A 抢占 B → A 运行

FreeRTOS 任务状态：

  • Running：正在 CPU 上执行（同一时刻只有一个）
  • Ready：准备好了，等 Scheduler 分配 CPU
  • Blocked：在等待某个事件（延时、Queue 数据、Semaphore）
  • Suspended：被手动暂停
3
创建第一个任务
20 分钟写代码

在 PlatformIO 项目里，ESP32 的 Arduino 框架已经内置 FreeRTOS，不需要额外安装任何库。

把 src/main.cpp 改成：

#include <Arduino.h>

// 任务函数：任务必须是无限循环（永远不能 return）
void taskBlink(void* parameter) {
  const int LED = 2;
  pinMode(LED, OUTPUT);

  for (;;) {                        // for(;;) 等同于 while(true)
    digitalWrite(LED, HIGH);
    vTaskDelay(500 / portTICK_PERIOD_MS);  // 延时 500ms，释放 CPU
    digitalWrite(LED, LOW);
    vTaskDelay(500 / portTICK_PERIOD_MS);
  }
}

void taskPrint(void* parameter) {
  for (;;) {
    Serial.println("Hello from Task 2!");
    vTaskDelay(1000 / portTICK_PERIOD_MS);  // 1000ms
  }
}

void setup() {
  Serial.begin(115200);

  // xTaskCreate(任务函数, 名字, 栈大小(bytes), 参数, 优先级, 句柄)
  xTaskCreate(taskBlink, "Blink", 1024, NULL, 1, NULL);
  xTaskCreate(taskPrint, "Print", 2048, NULL, 1, NULL);

  // setup() 结束后 FreeRTOS Scheduler 自动启动
}

void loop() {
  // Arduino 的 loop() 在 FreeRTOS 下其实也是一个任务
  // 当你用 FreeRTOS 时，最好把逻辑都放在任务里，loop() 留空
  vTaskDelay(1000 / portTICK_PERIOD_MS);
}
⚠️ 用 vTaskDelay 不是 delay()

delay(500) 会阻塞 CPU（其他任务无法运行）。vTaskDelay() 会让当前任务进入 Blocked 状态，调度器可以运行其他任务。FreeRTOS 里永远用 vTaskDelay。

💡 栈大小怎么选？

从 2048 字节开始，如果崩溃再加大。用 uxTaskGetStackHighWaterMark(NULL) 查看历史最小剩余栈（Step 8 详讲）。

4
Queue — 任务间传递数据
30 分钟

任务不能用全局变量直接通信（竞争条件），要用 Queue 安全传数据：

#include <Arduino.h>

QueueHandle_t dataQueue;   // Queue 句柄（全局变量）

// 生产者任务：模拟传感器读数（每 200ms 发一次数据）
void taskSensor(void* param) {
  int value = 0;
  for (;;) {
    value++;
    // xQueueSend(队列, 数据指针, 等待时间)
    // portMAX_DELAY = 永远等直到 Queue 有空间
    if (xQueueSend(dataQueue, &value, portMAX_DELAY) == pdPASS) {
      Serial.printf("Sensor sent: %d\n", value);
    }
    vTaskDelay(200 / portTICK_PERIOD_MS);
  }
}

// 消费者任务：从 Queue 取数据处理
void taskProcess(void* param) {
  int received;
  for (;;) {
    // xQueueReceive(队列, 存储指针, 等待时间)
    if (xQueueReceive(dataQueue, &received, portMAX_DELAY) == pdPASS) {
      Serial.printf("Processor got: %d, squared: %d\n", received, received * received);
    }
  }
}

void setup() {
  Serial.begin(115200);
  // xQueueCreate(队列长度, 每个元素大小)
  dataQueue = xQueueCreate(10, sizeof(int));

  if (dataQueue == NULL) {
    Serial.println("Queue creation failed!");
    return;
  }

  xTaskCreate(taskSensor,  "Sensor",  2048, NULL, 2, NULL);  // 优先级 2（高）
  xTaskCreate(taskProcess, "Process", 2048, NULL, 1, NULL);  // 优先级 1（低）
}

void loop() { vTaskDelay(1000 / portTICK_PERIOD_MS); }
Queue 工作原理

Queue 是 FIFO（先进先出）缓冲区。xQueueSend() 把数据复制进 Queue；xQueueReceive() 把数据取出并复制到你的变量。Queue 满时 Send 会阻塞，Queue 空时 Receive 会阻塞。

5
Mutex — 保护共享资源
20 分钟

当多个任务共用同一个资源（如 I2C 总线、全局缓冲区），要用 Mutex 确保每次只有一个任务访问：

#include <Arduino.h>

SemaphoreHandle_t i2cMutex;  // 保护 I2C 总线的 Mutex

// 模拟两个传感器任务共用 I2C
void taskSensorA(void* param) {
  for (;;) {
    // 请求 Mutex，最多等 100ms
    if (xSemaphoreTake(i2cMutex, pdMS_TO_TICKS(100)) == pdTRUE) {
      // ---- 临界区开始（只有我能运行到这里）----
      Serial.println("Sensor A: reading I2C...");
      vTaskDelay(50 / portTICK_PERIOD_MS);  // 模拟 I2C 通信耗时
      Serial.println("Sensor A: done");
      // ---- 临界区结束 ----
      xSemaphoreGive(i2cMutex);  // 释放 Mutex
    } else {
      Serial.println("Sensor A: couldn't get mutex, skipping");
    }
    vTaskDelay(500 / portTICK_PERIOD_MS);
  }
}

void taskSensorB(void* param) {
  for (;;) {
    if (xSemaphoreTake(i2cMutex, pdMS_TO_TICKS(100)) == pdTRUE) {
      Serial.println("Sensor B: reading I2C...");
      vTaskDelay(30 / portTICK_PERIOD_MS);
      Serial.println("Sensor B: done");
      xSemaphoreGive(i2cMutex);
    }
    vTaskDelay(300 / portTICK_PERIOD_MS);
  }
}

void setup() {
  Serial.begin(115200);
  i2cMutex = xSemaphoreCreateMutex();
  xTaskCreate(taskSensorA, "SensorA", 2048, NULL, 1, NULL);
  xTaskCreate(taskSensorB, "SensorB", 2048, NULL, 1, NULL);
}

void loop() { vTaskDelay(1000 / portTICK_PERIOD_MS); }
💡 Mutex vs Binary Semaphore 的区别

Mutex：有"所有权"，只有加锁的任务才能解锁，自动支持优先级继承（防 Priority Inversion）。用于保护共享资源。
Binary Semaphore：没有所有权，任何任务都可以 give/take，适合 ISR 通知任务。

6
Binary Semaphore — ISR 通知任务
20 分钟

中断服务程序（ISR）中不能做复杂操作，常见模式：ISR give Semaphore → 任务 take 后处理

#include <Arduino.h>

SemaphoreHandle_t buttonSem;
const int BUTTON_PIN = 0;   // ESP32 BOOT 按钮在 GPIO 0

// ISR：中断时只做一件事——give semaphore
void IRAM_ATTR buttonISR() {
  BaseType_t xHigherPriorityTaskWoken = pdFALSE;
  // ISR 里只能用 FromISR 版本的函数！
  xSemaphoreGiveFromISR(buttonSem, &xHigherPriorityTaskWoken);
  // 如果 give 唤醒了更高优先级任务，立即切换
  portYIELD_FROM_ISR(xHigherPriorityTaskWoken);
}

// 处理任务：等待 Semaphore，取到后处理
void taskButtonHandler(void* param) {
  for (;;) {
    // 永远等待（portMAX_DELAY），直到 ISR 给出 semaphore
    if (xSemaphoreTake(buttonSem, portMAX_DELAY) == pdTRUE) {
      Serial.printf("Button pressed! Free heap: %d bytes\n", esp_get_free_heap_size());
    }
  }
}

void setup() {
  Serial.begin(115200);
  buttonSem = xSemaphoreCreateBinary();

  // GPIO 0 下降沿触发中断
  pinMode(BUTTON_PIN, INPUT_PULLUP);
  attachInterrupt(digitalPinToInterrupt(BUTTON_PIN), buttonISR, FALLING);

  xTaskCreate(taskButtonHandler, "BtnHandler", 2048, NULL, 2, NULL);
}

void loop() { vTaskDelay(1000 / portTICK_PERIOD_MS); }
⚠️ ISR 规则

ISR 中只能调用带 FromISR 后缀的 FreeRTOS 函数（xQueueSendFromISR、xSemaphoreGiveFromISR）。普通版本会死锁。ISR 中不能 vTaskDelay、不能打印。

7
完整项目：ADC → Queue → Print
45 分钟项目

这是 Phase 2 的核心实践项目，整合所有概念：

  • Task A（优先级 2）：每 100ms 读一次 ADC（GPIO 34，接电位器或悬空）
  • Task B（优先级 1）：从 Queue 取数据，计算平均值，串口打印
#include <Arduino.h>

// 数据结构：包含时间戳和 ADC 值
struct SensorData {
  uint32_t timestamp_ms;
  int      adc_value;
};

QueueHandle_t adcQueue;
const int ADC_PIN = 34;   // ESP32 ADC1 CH6（GPIO 34 是输入专用）

// Task A：ADC 采集
void taskADC(void* param) {
  SensorData data;
  for (;;) {
    data.timestamp_ms = millis();
    data.adc_value    = analogRead(ADC_PIN);  // 0–4095（12-bit）

    // 发送到 Queue，等待最多 10ms
    if (xQueueSend(adcQueue, &data, pdMS_TO_TICKS(10)) != pdPASS) {
      Serial.println("ADC Queue full! Dropping sample.");
    }
    vTaskDelay(pdMS_TO_TICKS(100));  // 10Hz 采样
  }
}

// Task B：数据处理和打印
void taskPrint(void* param) {
  SensorData data;
  float     rolling_avg = 0;
  int       count = 0;

  for (;;) {
    if (xQueueReceive(adcQueue, &data, portMAX_DELAY) == pdPASS) {
      count++;
      rolling_avg = rolling_avg * 0.9f + data.adc_value * 0.1f;  // 指数移动平均
      Serial.printf("[%6lu ms] ADC=%4d  Avg=%.1f\n",
                    data.timestamp_ms, data.adc_value, rolling_avg);
    }
  }
}

void setup() {
  Serial.begin(115200);
  analogReadResolution(12);  // ESP32 默认 12-bit

  adcQueue = xQueueCreate(20, sizeof(SensorData));  // 20 个元素的 Queue
  if (!adcQueue) { Serial.println("Queue create failed!"); return; }

  xTaskCreate(taskADC,   "ADC",   2048, NULL, 2, NULL);
  xTaskCreate(taskPrint, "Print", 4096, NULL, 1, NULL);

  Serial.println("System started. Sampling at 10Hz...");
}

void loop() { vTaskDelay(pdMS_TO_TICKS(1000)); }
💡 测试方法

GPIO 34 悬空时 ADC 会读到随机值（正常现象）。连接一个 10kΩ 电位器（两端接 3.3V 和 GND，中间接 GPIO 34）可以看到平滑变化的值。

8
Stack Overflow 检测与调试
15 分钟

任务栈分配不够会导致随机崩溃，最难调试。FreeRTOS 提供内置检测：

// 在任何任务里加这行，查看历史最小剩余栈（word 为单位）
UBaseType_t watermark = uxTaskGetStackHighWaterMark(NULL);
Serial.printf("Task stack remaining: %u words (%u bytes)\n",
              watermark, watermark * 4);

// 建议：watermark < 50 words 时增大任务栈大小

在项目中加一个监控任务：

TaskHandle_t hADC, hPrint;  // 保存任务句柄

void taskMonitor(void* param) {
  for (;;) {
    Serial.println("=== Stack Monitor ===");
    Serial.printf("ADC task:   %u words left\n", uxTaskGetStackHighWaterMark(hADC));
    Serial.printf("Print task: %u words left\n", uxTaskGetStackHighWaterMark(hPrint));
    Serial.printf("Free heap:  %u bytes\n", esp_get_free_heap_size());
    vTaskDelay(pdMS_TO_TICKS(5000));  // 每 5 秒打印一次
  }
}

// 在 setup() 里：
xTaskCreate(taskADC,     "ADC",     2048, NULL, 2, &hADC);
xTaskCreate(taskPrint,   "Print",   4096, NULL, 1, &hPrint);
xTaskCreate(taskMonitor, "Monitor", 2048, NULL, 1, NULL);
⚠️ 常见崩溃原因

① 任务栈太小（用了大数组、递归、printf）② ISR 里调用了非 FromISR 函数 ③ 在任务里创建了大的局部数组（应改为静态或 heap 分配）

9
常见陷阱（省掉你几小时调试时间）
5 分钟阅读
  • 用 delay() 代替 vTaskDelay() → 整个 CPU 卡死，其他任务无法运行
  • 任务函数 return 了 → 必须是无限循环，return 会触发断言失败。如果想退出，调用 vTaskDelete(NULL)
  • Queue/Semaphore 未创建就使用 → 检查 xQueueCreate() 返回值是否为 NULL
  • 两个任务都 give Mutex → Mutex 有所有权，只有 take 的任务才能 give，否则行为未定义
  • ISR 中调用普通 FreeRTOS 函数 → 必须用 FromISR 版本
  • 优先级设错了 → FreeRTOS 数字越大优先级越高（与 Linux 相反）
