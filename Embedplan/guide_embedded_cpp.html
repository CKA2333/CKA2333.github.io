嵌入式 C++ vs 桌面 C++：哪些不能用
20 分钟

MCU 上的 C++ 是"受限 C++"。理解限制的原因，比背规则更重要。

❌ 通常不用 / 禁用
  • 异常处理（try/catch/throw）：异常表占 Flash，展开逻辑占 RAM，RT 不可预测
  • RTTI（dynamic_cast, typeid）：运行时类型信息，同样占空间
  • new / delete（堆动态分配）：碎片化，非确定性延迟，MCU RAM 本来就小
  • std::thread / std::mutex：桌面线程库，MCU 用 FreeRTOS 任务代替
  • std::string（频繁拼接）：内部有动态分配
  • 虚函数（在热路径中）：vtable 间接跳转，影响 icache
✅ 安全使用
  • constexpr / const：编译期计算，放入 Flash
  • 模板（无虚函数的静态多态）：零运行时开销
  • RAII（构造/析构管理资源）：栈上对象，生命周期确定
  • std::array：固定大小，无动态分配
  • std::optional / std::span（C++17）：现代 MCU SDK 已支持
  • Lambda / 函数对象：不捕获或只捕获值
🎯 面试常见问题

"为什么嵌入式不用异常处理？" → 两个理由：①代码体积膨胀（异常表），②中断/实时路径上延迟不确定。嵌入式用错误码（enum ErrorCode）或 std::optional 来处理错误。

内存模型：Stack / Heap / Static / Flash
30 分钟

MCU 的 RAM 通常 64 KB – 512 KB，Flash 通常 512 KB – 2 MB。知道每个变量在哪里，是嵌入式开发的基本功。

// ── 各内存区域示例 ─────────────────────────────────────────────

// 1. Flash（只读，断电不丢失）
const uint8_t font_table[] = { 0x00, 0x7E, ... };  // const 数组 → Flash
constexpr float PI = 3.14159f;                      // constexpr → Flash

// 2. .bss（初始化为0的全局变量，上电自动清零）
uint32_t event_count;                                // 全局 → .bss
static uint8_t uart_buffer[256];                    // static → .bss

// 3. .data（有初始值的全局变量，从 Flash 复制到 RAM）
uint32_t device_id = 0xDEADBEEF;                   // .data

// 4. Stack（函数内局部变量，自动管理，向下增长）
void process_sample() {
    float filtered[32];   // ← 128 字节占用栈空间！
    uint8_t temp_buf[64]; // ← 再多 64 字节
    // 函数返回时自动释放
}

// 5. Heap（new/malloc，嵌入式尽量避免）
uint8_t* dma_buf = new uint8_t[512];  // ← 嵌入式通常不这么做
// 改为静态分配：
static uint8_t dma_buf[512];            // 编译期确定大小
⚠️ 栈溢出（Stack Overflow）

嵌入式最常见的 crash 原因之一。MCU 上栈溢出不会报错，会直接静默地覆盖其他内存，导致随机 bug。FreeRTOS 提供 uxTaskGetStackHighWaterMark() 检测剩余栈空间。大数组用 static 放到 .bss，不要在函数内声明。

// ── 嵌入式内存使用规范 ─────────────────────────────────────────

// ❌ 危险：大局部数组
void bad_fft() {
    float buf[1024];  // 4KB 占栈！FreeRTOS 默认栈 512B-2KB
}

// ✅ 正确：用 static 或全局
static float fft_buf[1024];  // 放 .bss，编译期分配
void good_fft() {
    // 使用 fft_buf...
}

// ── 检查内存布局（GCC 链接后运行）─────────────────────────────
// arm-none-eabi-size build/firmware.elf
// 输出：text(Flash)  data(RAM初始值)  bss(RAM清零区)
//        48244         1024            8192

volatile 与中断处理（ISR）
40 分钟 · 面试高频！

volatile 告诉编译器："这个变量可能在你不知道的时候被改变，每次都要从内存读，不要缓存到寄存器。"

为什么 ISR 中需要 volatile

// ❌ 没有 volatile — 编译器可能把循环优化掉！
bool data_ready = false;

void UART_IRQHandler() {    // 中断处理函数
    data_ready = true;
}

void main_loop() {
    while (!data_ready) {}  // 编译器看到 data_ready 没被修改
    // → 可能优化为 while(true){}，永远等待！
}

// ✅ 加 volatile — 每次循环都从内存重新读取
volatile bool data_ready = false;

void UART_IRQHandler() {
    data_ready = true;    // ISR 写入
}

void main_loop() {
    while (!data_ready) {}  // 每次循环重新从 RAM 读取
    process_data();
}

完整的 ISR + 主循环通信模式

// ── UART 接收中断 → 环形缓冲区 → 主任务处理 ─────────────────────

constexpr size_t RX_BUF_SIZE = 256;

struct UartRxState {
    volatile uint8_t buf[RX_BUF_SIZE];
    volatile uint16_t head = 0;   // ISR 写入，主循环读取
    uint16_t tail = 0;            // 只有主循环写，不需要 volatile
} uart_rx;

// ISR：只做最少的工作！
extern "C" void USART1_IRQHandler() {
    if (USART1->SR & USART_SR_RXNE) {             // 接收寄存器非空
        uart_rx.buf[uart_rx.head] = USART1->DR;  // 读取数据
        uart_rx.head = (uart_rx.head + 1) % RX_BUF_SIZE;
    }
}

// 主循环：读取并处理
void uart_process() {
    while (uart_rx.tail != uart_rx.head) {   // 有新数据
        uint8_t byte = uart_rx.buf[uart_rx.tail];
        uart_rx.tail = (uart_rx.tail + 1) % RX_BUF_SIZE;
        handle_byte(byte);
    }
}
🚨 ISR 铁律

① ISR 必须尽量短（微秒级），只写 flag/buffer，不做复杂逻辑。②ISR 中不能调用 printf、malloc、或任何可能阻塞的函数。③ 访问 ISR 和主循环共享的多字节变量时，需要关中断或用原子操作（__atomic），否则读到中间状态。

原子访问（防止读到中间状态）

// 共享一个 32-bit 计数器（ISR 写，主循环读）
volatile uint32_t tick_count = 0;

void SysTick_Handler() {
    tick_count++;   // Cortex-M 上 32-bit 写是原子的，OK
}

// 但是 64-bit 不是原子的，需要保护：
volatile uint64_t timestamp_us = 0;

uint64_t get_timestamp_safe() {
    uint32_t primask = __get_PRIMASK();
    __disable_irq();            // 临时关中断
    uint64_t val = timestamp_us;  // 原子读取
    if (!primask) __enable_irq();  // 恢复
    return val;
}
4
状态机模式（State Machine）
30 分钟

嵌入式固件几乎所有逻辑都是状态机。LED 控制、通信协议解析、电池管理、设备连接流程 —— 全都用状态机来组织。

示例：BLE 连接状态机

// ── 状态定义 ────────────────────────────────────────────────
enum class BleState : uint8_t {    // 用 enum class + uint8_t，省内存
    IDLE,
    ADVERTISING,
    CONNECTING,
    CONNECTED,
    DISCONNECTING,
    ERROR
};

// ── 事件定义 ────────────────────────────────────────────────
enum class BleEvent : uint8_t {
    START_ADV,
    CONN_REQUEST,
    CONN_TIMEOUT,
    DISCONNECTED,
    ERROR_OCCURRED
};

// ── 状态机结构体 ─────────────────────────────────────────────
struct BleFsm {
    BleState state = BleState::IDLE;
    uint32_t state_enter_ms = 0;  // 进入当前状态的时间戳

    void handle(BleEvent ev) {
        switch (state) {
        case BleState::IDLE:
            if (ev == BleEvent::START_ADV)
                transition(BleState::ADVERTISING);
            break;

        case BleState::ADVERTISING:
            if (ev == BleEvent::CONN_REQUEST)
                transition(BleState::CONNECTING);
            else if (ev == BleEvent::CONN_TIMEOUT)
                transition(BleState::IDLE);
            break;

        case BleState::CONNECTING:
            if (ev == BleEvent::CONN_REQUEST)  // 连接成功
                transition(BleState::CONNECTED);
            else if (ev == BleEvent::CONN_TIMEOUT)
                transition(BleState::ADVERTISING);
            break;

        case BleState::CONNECTED:
            if (ev == BleEvent::DISCONNECTED)
                transition(BleState::IDLE);
            else if (ev == BleEvent::ERROR_OCCURRED)
                transition(BleState::ERROR);
            break;

        default: break;
        }
    }

    void transition(BleState next) {
        on_exit(state);           // 退出动作
        state = next;
        state_enter_ms = get_tick_ms();
        on_enter(next);           // 进入动作
    }

    void on_enter(BleState s) {
        switch (s) {
        case BleState::ADVERTISING: ble_start_adv(); break;
        case BleState::CONNECTED:   led_set_color(LED_BLUE); break;
        default: break;
        }
    }

    void on_exit(BleState s) {
        switch (s) {
        case BleState::ADVERTISING: ble_stop_adv(); break;
        default: break;
        }
    }

    // 超时检测（在主循环周期调用）
    void tick() {
        if (state == BleState::ADVERTISING) {
            if (get_tick_ms() - state_enter_ms > 5000)  // 5s 超时
                handle(BleEvent::CONN_TIMEOUT);
        }
    }
};
💡 为什么用 enum class 而不是 enum

enum class（C++11）不会隐式转换为整数，不会污染命名空间。BleState::IDLE 而不是 IDLE，避免与其他枚举值冲突。加上 : uint8_t 限制底层类型，节省内存（1字节 vs 默认4字节）。

环形缓冲区（Circular Buffer）完整实现
45 分钟 · 面试必考！

环形缓冲区（Ring Buffer / FIFO）是嵌入式中最常用的数据结构，用于 ISR → 主循环的数据传递，无需动态内存分配。

/**
 * 模板化环形缓冲区
 * 线程安全版本（一个写者 + 一个读者，无锁）
 */
template<typename T, size_t N>
class RingBuffer {
    static_assert((N & (N-1)) == 0, "N 必须是 2 的幂，让取模变位运算");
public:
    // 写入（ISR 或写者线程调用）
    bool push(const T& val) {
        size_t next = (head_ + 1) & (N - 1);  // 位运算取模（N是2的幂）
        if (next == tail_) return false;       // 满了，返回失败
        buf_[head_] = val;
        head_ = next;                           // 最后才更新 head（原子性）
        return true;
    }

    // 读取（主循环或读者线程调用）
    bool pop(T& out) {
        if (tail_ == head_) return false;      // 空的
        out = buf_[tail_];
        tail_ = (tail_ + 1) & (N - 1);
        return true;
    }

    bool empty() const { return tail_ == head_; }
    bool full()  const { return ((head_ + 1) & (N-1)) == tail_; }
    size_t size() const { return (head_ - tail_) & (N-1); }

private:
    T buf_[N];
    volatile size_t head_ = 0;  // ISR 写，volatile!
    volatile size_t tail_ = 0;  // 主循环写
};

// ── 使用示例 ─────────────────────────────────────────────────
RingBuffer<uint8_t, 256> uart_rx_fifo;  // 256字节 UART 接收 FIFO

void USART1_IRQHandler() {
    uart_rx_fifo.push(USART1->DR);      // ISR 写入
}

void main_task() {
    uint8_t byte;
    while (uart_rx_fifo.pop(byte)) {
        protocol_parse(byte);            // 主循环处理
    }
}
🔍 为什么 N 必须是 2 的幂

index % N → index & (N-1)：位运算替代取模，在没有硬件除法的 MCU 上速度快 10–100 倍。N=256 时，& 0xFF 比 % 256 在 Cortex-M0 上快约 20 倍。

定点数算术（Fixed-Point）
30 分钟

Cortex-M0/M0+ 没有 FPU（浮点运算单元）。一次 float 加法 = 软件模拟 = 约 50–100 个时钟周期 vs. 整数加法 1 个周期。定点数用整数表示小数，在无 FPU 的 MCU 上速度快 50–100 倍。

/**
 * Q15 定点数格式：1位符号 + 0位整数 + 15位小数
 * 范围：-1.0 ~ +0.99997，精度：1/32768 ≈ 0.00003
 * 表示方式：value_int16 / 32768.0 = 实际值
 */
using q15_t = int16_t;
constexpr int32_t Q15_SCALE = 32768;  // = 1 << 15

// float → Q15 转换（离线做，或在初始化时做一次）
constexpr q15_t f2q15(float f) {
    return (q15_t)(f * Q15_SCALE);
}

// Q15 加法（和普通整数加法一样，O(1)）
q15_t q15_add(q15_t a, q15_t b) {
    int32_t result = (int32_t)a + b;
    // 饱和截断（防止 overflow 绕回）
    if (result >  32767) return  32767;
    if (result < -32768) return -32768;
    return (q15_t)result;
}

// Q15 乘法（结果是 Q30，需要右移 15 位才回到 Q15）
q15_t q15_mul(q15_t a, q15_t b) {
    int32_t product = (int32_t)a * b;
    return (q15_t)(product >> 15);  // 右移 15 = 除以 32768
}

// ── 实际应用：一阶低通滤波器（IIR）─────────────────────────
// float 版：out = alpha * in + (1-alpha) * out_prev
// Q15 版：完全用整数运算

struct LpfQ15 {
    q15_t alpha;      // 滤波系数，如 0.1 → f2q15(0.1f)
    q15_t alpha_inv;  // 1 - alpha
    q15_t state;

    LpfQ15(float a) : alpha(f2q15(a)), alpha_inv(f2q15(1.0f - a)), state(0) {}

    q15_t update(q15_t input) {
        // out = alpha * in + alpha_inv * state
        state = q15_add(q15_mul(alpha, input),
                         q15_mul(alpha_inv, state));
        return state;
    }
};

// 使用：
LpfQ15 filter(0.1f);  // α = 0.1（截止频率由采样率决定）
q15_t raw = f2q15(0.75f);  // ADC 读值（已归一化到 -1~1）
q15_t filtered = filter.update(raw);
💡 CMSIS-DSP 库

ARM 提供了 CMSIS-DSP 库，包含优化好的 Q15/Q31 运算函数（FFT、FIR 滤波器、矩阵运算）。Nordic nRF5 SDK 和 STM32 HAL 都包含这个库。实际工作中直接用，不用手写。但面试时能说明白定点数原理，会加分。

RAII 在嵌入式的正确使用
20 分钟

RAII（Resource Acquisition Is Initialization）= 构造时获取资源，析构时释放。嵌入式里最常用于：关中断区间、SPI/I2C 总线锁、GPIO 操作。

// ── 示例 1：关中断保护区（RAII）────────────────────────────
class CriticalSection {
public:
    CriticalSection()  { primask_ = __get_PRIMASK(); __disable_irq(); }
    ~CriticalSection() { if (!primask_) __enable_irq(); }
    CriticalSection(const CriticalSection&) = delete;  // 不允许拷贝
private:
    uint32_t primask_;
};

void update_shared_data(uint32_t val) {
    CriticalSection cs;      // 构造时关中断
    shared_counter += val;   // 安全操作
}                            // 析构时自动恢复中断，即使提前 return 也安全

// ── 示例 2：SPI 片选（RAII）───────────────────────────────
class SpiSelect {
public:
    SpiSelect(GPIO_TypeDef* port, uint16_t pin)
        : port_(port), pin_(pin) {
        port_->BSRR = pin_ << 16;  // CS 低（片选激活）
    }
    ~SpiSelect() {
        port_->BSRR = pin_;          // CS 高（片选释放）
    }
    SpiSelect(const SpiSelect&) = delete;
private:
    GPIO_TypeDef* port_;
    uint16_t pin_;
};

void spi_read_sensor(uint8_t* buf, size_t len) {
    SpiSelect sel(GPIOA, GPIO_PIN_4);  // 拉低 CS
    spi_transfer(buf, len);
}  // sel 析构，自动拉高 CS，即使 spi_transfer 抛出错误也安全
💡 为什么 RAII 在嵌入式比 try/finally 更好

C++ 没有 finally（Java 才有）。RAII 用析构函数实现了同样的"无论如何都执行清理"的效果，而且完全在编译期确定，零运行时开销，不依赖异常机制。Nordic SDK 的 nrfx 驱动层大量使用这个模式。

面试题精选 + 答题框架
随时复习

高频面试题

Q1 · 什么时候用 volatile？

答：三种场景：① ISR 和主循环共享的变量；② 内存映射硬件寄存器（volatile uint32_t* const REG = (uint32_t*)0x40000000）；③ 等待硬件操作完成的 busy-wait 标志位。
不是银弹：volatile 只防止编译器优化读写，不保证多核原子性（多核要用 std::atomic 或内存屏障）。

Q2 · 什么是内存对齐？为什么重要？

答：32-bit MCU 访问 4 字节对齐的地址最高效；访问未对齐地址（如奇数地址的 uint32_t）在 Cortex-M0 上会 HardFault。编译器默认会对结构体做对齐填充，可以用 __attribute__((packed)) 禁用（但会影响性能，只在协议包解析时用）。

Q3 · 解释 DMA 是什么以及为什么嵌入式要用它

答：DMA（Direct Memory Access）允许外设直接向 RAM 写数据，不经过 CPU。比如 ADC 采样 1kHz，如果每次都用 CPU 读，CPU 就没时间做其他事了。用 DMA：ADC 把数据自动搬到 buf[]，传输完成后才产生中断通知 CPU 处理，CPU 利用率大幅提升。关键：DMA 源/目的地址必须 4 字节对齐，buffer 不能用栈变量（栈上地址不稳定，且 DMA 用物理地址）。

Q4 · 如何调试栈溢出？

答：① FreeRTOS 的 uxTaskGetStackHighWaterMark() 可查各任务最小剩余栈空间。② 在 linker script 里用 canary pattern 填充栈底，溢出时触发 HardFault 并在 handler 里打印状态。③ ARM Cortex-M4 有 MPU，可设置栈保护区，访问时自动触发 fault。
